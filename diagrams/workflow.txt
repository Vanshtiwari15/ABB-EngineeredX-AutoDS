    │    3. FastText                                           │
    │       ├─ Library: fasttext                               │
    │       ├─ Pros: Fast, multilingual, efficient            │
    │       ├─ Cons: Less flexible, lower accuracy            │
    │       └─ Params: {epoch: 25, lr: 0.5, wordNgrams: 2}   │

    │ B. PREPROCESSING PIPELINE                               │
    │    1. Text Cleaning                                      │
    │       • Remove special characters                        │

    │       • Word or subword tokens                           │
    │    3. Lowercasing                                        │
    │       • Convert all text to lowercase                    │
    │    4. Stopword Removal                                   │
    │       • Remove common words (NLTK list)                  │
    │    5. Vectorization                                      │
    │       • TF-IDF or word embeddings                        │
    │                                                           │
    │ C. EVALUATION METRICS                                    │
    │    1. Accuracy                                           │
    │       Formula: Correct predictions / Total               │
    │       Use case: General classification tasks             │
    │                                                           │
    │    2. F1-Score                                           │
    │       Formula: 2*(Precision*Recall)/(P+R)               │
    │       Use case: Token classification                     │
    │                                                           │
    │    3. Perplexity                                         │
    │       Formula: 2^(-1/N * Σ log(P(word)))                │
    │       Use case: Language models                          │

                    MODEL SELECTION LOGIC BY TASK TYPE

    ┌─────────────────────────────────────────────────────────────┐
    │ CLASSIFICATION TASK                                         │
    ├─────────────────────────────────────────────────────────────┤
    │ IF dataset_size < 10K                                        │
    │   THEN model_1 = RandomForest  (good generalization)        │
    │        model_2 = LogisticRegression  (baseline)              │
    │        model_3 = SVM  (flexible)                             │
    │                                                              │
    │ ELIF dataset_size 10K-100K                                   │
    │   THEN model_1 = XGBoost  (high accuracy)                   │
    │        model_2 = RandomForest  (robust)                      │
    │        model_3 = LightGBM  (fast)                            │
    │                                                              │
    │ ELIF dataset_size > 100K                                     │
    │   THEN model_1 = XGBoost  (scalable)                        │
    │        model_2 = Neural Networks  (complex patterns)         │
    │        model_3 = SGDClassifier  (online learning)            │
    └─────────────────────────────────────────────────────────────┘

    ┌─────────────────────────────────────────────────────────────┐
    │ REGRESSION TASK                                             │
    ├─────────────────────────────────────────────────────────────┤
    │ IF linear_relationship                                       │
    │   THEN model_1 = LinearRegression  (interpretable)           │
    │        model_2 = Ridge  (regularization)                     │
    │        model_3 = Lasso  (feature selection)                  │
    │                                                              │
    │ ELIF non_linear_patterns                                     │
    │   THEN model_1 = XGBoostRegressor  (high accuracy)          │
    │        model_2 = RandomForestRegressor  (robust)             │
    │        model_3 = KernelRidge  (flexibility)                  │
    │                                                              │
    │ IF outliers_present                                          │
    │   THEN use_robust_model = RandomForest / SVR                 │
    │        avoid = LinearRegression                              │
    └─────────────────────────────────────────────────────────────┘

    ┌─────────────────────────────────────────────────────────────┐
    │ NLP TASK                                                    │
    ├─────────────────────────────────────────────────────────────┤
    │ IF high_accuracy_needed                                      │
    │   THEN model_1 = BERT + FineTuning  (state-of-art)          │
    │        model_2 = RoBERTa  (similar performance)              │
    │        model_3 = DistilBERT  (faster)                        │
    │                                                              │
    │ ELIF speed_important                                         │
    │   THEN model_1 = FastText  (very fast)                      │
    │        model_2 = TF-IDF + SVM  (classical)                   │
    │        model_3 = Naive Bayes  (simple)                       │
    │                                                              │
    │ IF dataset_small                                             │
    │   THEN avoid_deep_learning = True                            │
    │        use_classical = TF-IDF + SVM / FastText               │
    └─────────────────────────────────────────────────────────────┘

    ┌─────────────────────────────────────────────────────────────┐
    │ TIME SERIES TASK                                            │
    ├─────────────────────────────────────────────────────────────┤
    │ IF univariate AND stationary                                │
    │   THEN model_1 = ARIMA  (statistical approach)              │
    │        model_2 = ExponentialSmoothing  (trend/seasonal)     │
    │        model_3 = AutoRegression  (simple)                    │
    │                                                              │
    │ ELIF multivariate OR complex_patterns                        │
    │   THEN model_1 = LSTM  (RNN, temporal dependencies)         │
    │        model_2 = Prophet  (handles seasonality)              │
    │        model_3 = ARIMA-GARCH  (volatility)                   │
    │                                                              │
    │ IF long_sequences                                            │
    │   THEN prefer_transformer = Temporal Fusion Transformer     │
    └─────────────────────────────────────────────────────────────┘
